<script>
    function show(id) {
        document.querySelectorAll('.section').forEach(s => s.style.display = 'none');
        document.getElementById(id).style.display = 'block';
        localStorage.setItem('tab_activa', id);
    }

    function toggleForm(id) {
        const f = document.getElementById(id);
        f.style.display = (f.style.display === 'block') ? 'none' : 'block';
    }

    function filterTable(inputId, tableId) {
        const filter = document.getElementById(inputId).value.toLowerCase();
        const tr = document.getElementById(tableId).getElementsByTagName("tr");
        for (let i = 1; i < tr.length; i++) {
            let visible = false;
            const td = tr[i].getElementsByTagName("td");
            for (let j = 0; j < td.length - 1; j++) {
                if (td[j] && td[j].innerText.toLowerCase().indexOf(filter) > -1) { visible = true; }
            }
            tr[i].style.display = visible ? "" : "none";
        }
    }

    // Lógica mejorada: Evita duplicados en la lista de materiales
    function addM() {
        const s = document.getElementById('s-mat');
        const c = document.getElementById('c-mat');
        
        if(!s.value) return;

        // Comprobación de duplicados
        const hiddenInputs = document.querySelectorAll('#l-m input[name="m_ids[]"]');
        let existe = false;
        hiddenInputs.forEach(input => {
            if(input.value === s.value) {
                existe = true;
            }
        });

        if(existe) {
            alert("Este material ya ha sido agregado a la lista del vale actual.");
            return;
        }

        const html = `<tr>
            <td>${s.options[s.selectedIndex].dataset.n}<input type="hidden" name="m_ids[]" value="${s.value}"></td>
            <td>${c.value}<input type="hidden" name="m_cants[]" value="${c.value}"></td>
            <td><button type="button" onclick="this.parentElement.parentElement.remove()" style="color:red; border:none; background:none; cursor:pointer; font-weight:bold;">X</button></td>
        </tr>`;
        
        document.getElementById('l-m').insertAdjacentHTML('beforeend', html);
        document.getElementById('t-m').style.display = 'table';
        
        // Limpiar selección
        s.value = ""; 
        c.value = 1;
    }

    window.onload = () => {
        const urlParams = new URLSearchParams(window.location.search);
        const tab = urlParams.get('tab') || localStorage.getItem('tab_activa') || 'n_vale';
        show(tab);
    }
</script>
